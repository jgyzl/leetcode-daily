class Solution {
    public int firstMissingPositive(int[] nums) {
        int len = nums.length;

        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < len; i++) {
            set.add(nums[i]);
        }
        for (int i = 1; i <= len; i++) {
            if (!set.contains(i)){
                return i;
            }
        }
        return len+1;
    }
